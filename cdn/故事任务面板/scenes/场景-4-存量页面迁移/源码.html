📄 内联消除度量脚本

#!/usr/bin/env bash
# migrate-metrics.sh — 度量页面内联消除率
set -euo pipefail

PAGE="${1:?usage: migrate-metrics.sh page.html}"

STYLE_BLOCKS=$(grep -c '<style>' "$PAGE" || echo 0)
INLINE_LINES=$(awk '/<style>/,/<\/style>/' "$PAGE" | wc -l)
HARDCODED_COLORS=$(grep -oE '#[0-9a-fA-F]{3,8}' "$PAGE" | wc -l)
YRY_TAGS=$(grep -oE '<yry-[a-z-]+' "$PAGE" | sort -u | wc -l)

echo "📊 迁移度量: $PAGE"
echo "  <style> 块数:        $STYLE_BLOCKS  (目标 ≤ 1)"
echo "  内联 style 行数:     $INLINE_LINES  (目标 ≤ 50)"
echo "  硬编码颜色:          $HARDCODED_COLORS  (目标 ≤ 0)"
echo "  yry-* 自定义元素:    $YRY_TAGS  (目标 ≥ 5)"

# 退出码: 任一指标超阈值即失败
[ "$STYLE_BLOCKS" -le 1 ] && \
[ "$INLINE_LINES" -le 50 ] && \
[ "$HARDCODED_COLORS" -le 0 ] || exit 1

📄 保留页面专属 CSS 判断

// 判断某段 CSS 是否应保留为页面专属
function shouldKeepAsPageCSS(rule) {
  // 1. 页面特有布局 (如 5 列相册网格)
  if (/grid-template-columns:\s*repeat\(5/.test(rule)) return true;

  // 2. 页面特有动画 (首屏入场序列)
  if (/@keyframes\s+intro-/.test(rule)) return true;

  // 3. 特定间距微调 (覆盖设计令牌)
  if (/margin-top:\s*13px/.test(rule)) return true;

  // 4. A/B 测试变体 (临时)
  if (/\.variant-[A-Z]\b/.test(rule)) return true;

  // 其他情况: 应迁移到组件
  return false;
}
