# 1. 加载链可用性 (5 步全部 HTTP 200)
curl -sI https://cdn.jsdelivr.net/npm/yry-cdn@1.2.0/shared/index.css | head -1
curl -sI https://cdn.jsdelivr.net/npm/yry-cdn@1.2.0/theme/index.css   | head -1
curl -sI https://unpkg.com/vue@3/dist/vue.global.prod.js                | head -1

# 2. SRI 哈希生成 (sha384)
openssl dgst -sha384 -binary shared/index.js | openssl base64 -A

# 3. CSP 检查 (应无 unsafe-inline)
curl -sI https://cdn.jsdelivr.net/npm/yry-cdn@1.2.0/index.html | grep -i content-security

# 4. 压缩与 HTTP/2 检查
curl -sI -H 'Accept-Encoding: br,gzip' https://cdn.jsdelivr.net/npm/yry-cdn@1.2.0/shared/index.js | grep -E 'content-encoding|alt-svc'
