📄 组件文件结构 (三文件拆分)

yry-{component}/
├── index.html    # 模板源 (Vue template) + Demo 预览
├── index.js      # Loader: fetch 模板 → defineCustomElement → 派发 ready 事件
└── index.css     # 组件样式 (使用 --yry-* 设计令牌 + fallback)

三文件拆分让模板、逻辑、样式各司其职，便于独立测试与按需加载。

📄 组件注册模式 (以 yry-item-card 为例)

// yry-item-card/index.js
fetch('yry-item-card/index.html')
  .then(r => r.text())
  .then(html => {
    const template = html.match(
      /<template[^>]*>([\s\S]*?)<\/template>/)[1];
    const YryItemCard = Vue.defineCustomElement({
      template,
      props: {
        icon: String, name: String, badge: String,
        desc: String, tags: Array, meta: String,
        links: Array, nameHref: String, nameTarget: String,
        iconModifier: { type: String, default: 'skill' }
      },
      emits: ['item-card-ready']
    });
    customElements.define('yry-item-card', YryItemCard);
    document.dispatchEvent(
      new CustomEvent('yry-item-card-ready')
    );
  });

📄 页面消费组件 (双设策略)

<!-- HTML -->
<yry-item-card id="card-1"></yry-item-card>

<script>
  // 第 1 次: 立即赋值 (元素已在 DOM, 组件可能尚未定义)
  var el = document.getElementById('card-1');
  el.icon = '🃏';
  el.name = '组件名';
  el.desc = '描述';
  el.tags = ['tag1', 'tag2'];

  // 第 2 次: 监听 ready 事件 (覆盖组件晚于元素出现的场景)
  document.addEventListener('yry-item-card-ready', function () {
    el.icon = '🃏';   // 重新赋值确保 props 已被 Vue 接收
    el.name = '组件名';
  });
</script>

双设策略覆盖两种时序：组件先就绪时第一次赋值生效；元素先于组件就绪时第二次赋值兜底。

// copyCmd: 非 https 时退回 execCommand
function copyCmd(text, el) {
  if (navigator.clipboard && window.isSecureContext) {
    navigator.clipboard.writeText(text);
  } else {
    var ta = document.createElement('textarea');
    ta.value = text;
    document.body.appendChild(ta);
    document.execCommand('copy');
    document.body.removeChild(ta);
  }
}

// fetchData: 指数退避重试
function fetchData(url, options, retries) {
  retries = retries || 3;
  return fetch(url, options).catch(function (err) {
    if (retries <= 0) throw err;
    return new Promise(function (resolve) {
      setTimeout(function () {
        resolve(fetchData(url, options, retries - 1));
      }, Math.pow(2, 4 - retries) * 1000);
    });
  });
}
