📄 发布前检查清单脚本

#!/usr/bin/env bash
# publish-prep.sh — 发布前 6 项检查
set -euo pipefail

# 1. 确认版本号正确递增
VERSION=$(node -e "console.log(require('./package.json').version)")
echo "📦 当前版本: $VERSION"

# 2. 确认 files 白名单无遗漏
echo "📋 files 白名单预览:"
npm pack --dry-run 2>&1 | grep -E '^npm notice' | head -20

# 3. 检查包体积
SIZE=$(npm pack --dry-run 2>&1 | tail -1 | grep -oE '[0-9]+\.?[0-9]*[km]?B')
echo "📏 包体积: $SIZE (目标 < 5MB)"

# 4. 更新 CHANGELOG.md
if ! grep -q "## \[$VERSION\]" CHANGELOG.md; then
  echo "⚠️  CHANGELOG.md 未含 $VERSION 段落"
  exit 1
fi

# 5. 确认所有文件已提交
if [ -n "$(git status --porcelain)" ]; then
  echo "⚠️  working tree 不干净:"
  git status --short
  exit 1
fi

# 6. 运行测试
echo "🧪 运行测试..."
npm test

echo "✅ 发布前检查全部通过"

📄 发布后验证脚本

#!/usr/bin/env bash
# publish-verify.sh — 发布后 5 步验证
set -euo pipefail
V="${1:?usage: publish-verify.sh 1.3.0}"

# 1. 验证 npm 安装
mkdir -p /tmp/yry-verify-$V && cd /tmp/yry-verify-$V
npm install yry-cdn@$V
ls node_modules/yry-cdn/  # 期望: shared theme theme-mono fonts yry-* *.md

# 2. 验证 jsDelivr 5 步加载链
for path in shared/index.css theme/index.css shared/index.js; do
  code=$(curl -sI "https://cdn.jsdelivr.net/npm/yry-cdn@$V/$path" | head -1 | grep -oE '[0-9]{3}')
  [ "$code" = "200" ] || echo "✗ jsDelivr $path: $code"
done

# 3. 验证 unpkg 可用
curl -sI "https://unpkg.com/yry-cdn@$V/shared/index.css" | head -1

# 4. SRI 哈希生成
echo "🔐 SRI 哈希:"
openssl dgst -sha384 -binary shared/index.js | openssl base64 -A
echo

# 5. dist-tag 检查
echo "🏷️  dist-tag:"
npm dist-tag ls yry-cdn

📄 版本策略决策树 (伪代码)

function decideVersionBump(commits) {
  // 检查是否有破坏性变更
  if (commits.some(c => c.type === 'BREAKING')) {
    return { segment: 'MAJOR', channel: 'latest', rollback: '30d' };
  }
  // 检查是否有新功能
  if (commits.some(c => c.type === 'feat')) {
    return { segment: 'MINOR', channel: 'latest', rollback: '14d' };
  }
  // 检查是否有修复
  if (commits.some(c => c.type === 'fix' || c.type === 'perf' || c.type === 'docs')) {
    return { segment: 'PATCH', channel: 'latest', rollback: '7d' };
  }
  // 不需要发布
  return { segment: null, channel: null, rollback: null };
}
