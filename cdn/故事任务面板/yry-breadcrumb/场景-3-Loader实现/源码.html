代码块 1 -- IIFE 加载器框架

Loader 使用 IIFE（立即执行函数表达式）包装，确保独立作用域，'use strict' 严格模式。

JSindex.js — IIFE 框架
// cdn/yry-breadcrumb/index.js -- IIFE 框架
(function () {
  'use strict';

  // 1. Vue 环境检测
  if (!window.Vue) {
    console.warn('[YryBreadcrumb] Vue 3 未加载，跳过注册');
    return;
  }

  // 2. 脚本路径解析
  var script = document.currentScript;
  var scriptUrl = new URL(script.src, window.location.href);
  var templateUrl = new URL('index.html', scriptUrl).href;

  // 3. 超时计时器（5s）
  var timedOut = false;
  var timeoutId = setTimeout(function () {
    timedOut = true;
    console.error('[YryBreadcrumb] 模板加载超时（5s）');
  }, 5000);

  // ... fetch + DOMParser 逻辑
})();

代码块 2 -- fetch + DOMParser 核心逻辑

核心加载流程：fetch 获取 index.html 文本，DOMParser 解析，提取 template。

JSindex.js — fetch + DOMParser
// cdn/yry-breadcrumb/index.js -- fetch + DOMParser
  fetch(templateUrl, { credentials: 'same-origin' })
    .then(function (response) {
      if (!response.ok) {
        throw new Error('HTTP ' + response.status + ': ' + response.statusText);
      }
      return response.text();
    })
    .then(function (htmlText) {
      // 超时则丢弃结果
      if (timedOut) return;
      clearTimeout(timeoutId);

      // DOMParser 解析
      var parser = new DOMParser();
      var doc = parser.parseFromString(htmlText, 'text/html');

      // 提取模板
      var tpl = doc.getElementById('yry-breadcrumb-tpl');
      if (!tpl) {
        console.error('[YryBreadcrumb] 模板元素 #yry-breadcrumb-tpl 未找到');
        return;
      }

      // 注册组件定义
      window.YryBreadcrumb = {
        name: 'YryBreadcrumb',
        props: {
          items: { type: Array, required: true },
          ariaLabel: { type: String, default: '面包屑导航' },
          separator: { type: String, default: '/' }
        },
        template: tpl.innerHTML
      };

      // 派发 ready 事件
      document.dispatchEvent(new CustomEvent('yry-breadcrumb-ready'));
    })

ready 事件机制和完整的错误降级处理。

JSindex.js — CustomEvent + 错误处理
// cdn/yry-breadcrumb/index.js -- CustomEvent + 错误处理
    .catch(function (err) {
      clearTimeout(timeoutId);
      console.error('[YryBreadcrumb] 模板加载失败:', err.message || err);
      // 不注册组件，页面继续正常运行（降级）
    });

// --- 页面侧 mount 调用 ---
function mountBreadcrumb() {
  if (!window.Vue || !window.YryBreadcrumb) return;
  Vue.createApp(window.YryBreadcrumb, {
    ariaLabel: '面包屑导航',
    items: [
      { label: 'YryBreadcrumb', href: '../../../index.html', icon: '🍞' },
      { label: '任务故事', href: '../README.md' },
      { label: '场景 3 · Loader 实现' },
      { label: '源码', icon: '📄' }
    ]
  }).mount('#breadcrumb-app');
}

// 双路径挂载：已 ready 直接挂，否则监听 ready 事件
if (window.YryBreadcrumb) {
  mountBreadcrumb();
} else {
  document.addEventListener('yry-breadcrumb-ready', mountBreadcrumb, { once: true });
}
