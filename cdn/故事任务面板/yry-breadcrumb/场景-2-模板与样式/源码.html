1. Template 源码 (v-for / v-if / :class)

来自 cdn/yry-breadcrumb/index.html 中 <script type="text/x-template"> 块。

HTMLindex.html
<script type="text/x-template" id="yry-breadcrumb-template">
  <nav class="breadcrumb" :aria-label="ariaLabel">
    <template v-for="(item, i) in items" :key="i">
      <!-- 分隔符：第一个元素前不渲染 -->
      <span v-if="i > 0" class="bc-sep" aria-hidden="true">/</span>

      <!-- 链接项：有 href 时渲染 a 标签 -->
      <a v-if="item.href"
         :href="item.href"
         :class="{ 'bc-link-icon': item.icon }">
        <template v-if="item.icon">{{ item.icon }}&nbsp;</template>
        {{ item.label }}
      </a>

      <!-- 当前项：无 href 时为纯文本 -->
      <span v-else
            class="bc-current"
            :aria-current="i === items.length - 1 ? 'page' : null">
        <template v-if="item.icon">{{ item.icon }}&nbsp;</template>
        {{ item.label }}
      </span>
    </template>
  </nav>
</script>

3. Demo 挂载示例代码

来自各场景 HTML 文件中的 breadcrumb mount 脚本。

JSmount.js
// === 标准挂载模式（dogfooding：用组件渲染自身导航）===

function mountBreadcrumb() {
  // 等待 Vue 3 和 YryBreadcrumb 均就绪
  if (!window.Vue || !window.YryBreadcrumb) return;

  Vue.createApp(window.YryBreadcrumb, {
    ariaLabel: '面包屑导航',
    items: [
      { label: 'YryBreadcrumb', href: '../../../index.html', icon: '🍞' },
      { label: '任务故事',       href: '../README.md' },
      { label: '场景 2 · 模板与样式' },
      { label: '关键源码',       icon: '📄' }
    ]
  }).mount('#breadcrumb-app');
}

// 双保险：同步尝试 + 异步事件监听
if (window.YryBreadcrumb) {
  mountBreadcrumb();
} else {
  document.addEventListener('yry-breadcrumb-ready', mountBreadcrumb, { once: true });
}
