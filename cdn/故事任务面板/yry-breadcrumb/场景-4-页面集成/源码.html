1. 页面集成模板 (5 步加载模式)

任意目标页面引入 YryBreadcrumb 的标准模板 — 4 文件引用 + mount 脚本。

HTMLintegration-template.html
<!-- ============ 步骤 1: CSS (head 首位) ============ -->
<link rel="stylesheet" href="path/to/cdn/yry-breadcrumb/index.css">
<!-- ============ 步骤 2: Vue 3 运行时 ============ -->
<script src="https://unpkg.com/vue@3/dist/vue.global.prod.js"></script>

<!-- ============ 步骤 3: 挂载点 (body 合适位置) ============ -->
<div id="breadcrumb-app"></div>

<!-- ============ 步骤 4: Loader 脚本 ============ -->
<script src="path/to/cdn/yry-breadcrumb/index.js"></script>

<!-- ============ 步骤 5: mount 执行 ============ -->
<script>
  function mountBreadcrumb() {
    if (!window.Vue || !window.YryBreadcrumb) return;
    Vue.createApp(window.YryBreadcrumb, {
      ariaLabel: '面包屑导航',
      items: [ /* ... */ ]
    }).mount('#breadcrumb-app');
  }
  // 缓存命中直接挂载，否则监听 ready 事件
  if (window.YryBreadcrumb) mountBreadcrumb();
  else document.addEventListener('yry-breadcrumb-ready', mountBreadcrumb, { once: true });
</script>

2. Mount 脚本示例

典型的 mount 调用，包含 ariaLabel 和无障碍属性。

JSmount.js
// 本页实际使用的 mount 脚本
function mountBreadcrumb() {
  if (!window.Vue || !window.YryBreadcrumb) return;
  Vue.createApp(window.YryBreadcrumb, {
    ariaLabel: '面包屑导航',
    items: [
      { label: 'YryBreadcrumb', href: '../../../index.html', icon: '🍞' },
      { label: '任务故事', href: '../README.md' },
      { label: '场景 4 · 页面集成' },
      { label: '关键源码', icon: '📄' }
    ]
  }).mount('#breadcrumb-app');
}
if (window.YryBreadcrumb) mountBreadcrumb();
else document.addEventListener('yry-breadcrumb-ready', mountBreadcrumb, { once: true });
