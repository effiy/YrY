YryProgressBar · 场景 1/5 · Props 定义 · 双模式判定

Props API 定义 (index.js)

props: {
  done: { type: Number, default: 0 },
  total: { type: Number, default: 0 },
  label: { type: String, default: '进度' }
}

双模式判定逻辑 (mounted)

mounted: function () {
  var el = this.$el;
  // 判定: 有 done 或 total 属性 → 显式模式，否则 → 滚动追踪
  if (el.hasAttribute('done') || el.hasAttribute('total')) return;
  this._scrollMode = true;
  // ... 绑定 scroll + resize listeners
}

进度计算 (computed pct)

computed: {
  pct: function () {
    var d = this._scrollMode ? this._val.done : this.done;
    var t = this._scrollMode ? this._val.total : this.total;
    return t > 0 ? Math.round((d / t) * 100) : 0;
  }
}
