1. cdn/package.json 字段

npm 包配置文件，定义包名、版本、发布文件列表。files[] 必须包含全部 4 个 yry-breadcrumb 核心文件路径。

// cdn/package.json — npm 集成配置
{
  "name": "yry-cdn",
  "version": "1.1.0",
  "description": "YrY CDN 静态资源库 — 共享 CSS/JS/组件",
  "main": "shared.js",
  "files": [
    "shared.css",
    "shared.js",
    "theme.css",
    "theme-mono.css",
    "fonts.css",
    "fonts/*.woff2",
    "yry-checklist.css",

    "yry-breadcrumb/任务故事/README.md",

2. npm publish 发布流程

从语义化版本升级到 npm 发布的完整命令序列。

# Step 1: 确认工作区清洁
git status

# Step 2: 升级版本号 (minor — 新增组件无破坏性变更)
cd cdn
npm version minor
# → 输出: v1.1.0
# → 自动 commit: "1.1.0" + git tag v1.1.0

# Step 3: 打包验证
npm pack --dry-run
# → 确认 npm notice 列表中包含:
#   yry-breadcrumb/index.html
#   yry-breadcrumb/index.js
#   yry-breadcrumb/index.css
#   yry-breadcrumb/任务故事/README.md

# Step 4: 生成 tarball 并解包验证
npm pack
tar -tzf yry-cdn-1.1.0.tgz | grep yry-breadcrumb

# Step 5: 推送到 Git 远程 (含 tag)
git push origin main --tags

# Step 6: 发布到 npm registry
npm publish
# → 灰度发布: npm publish --tag next
# → 稳定发布: npm dist-tag add yry-cdn@1.1.0 latest

3. 版本号升级脚本示例

自动化语义化版本升级及 tag 创建的辅助脚本。

#!/usr/bin/env node
// scripts/bump-version.mjs — 语义化版本升级工具

import { readFileSync, writeFileSync } from 'fs';
import { execSync } from 'child_process';

const PKG_PATH = './cdn/package.json';
const bumpType = process.argv[2] || 'patch'; // major | minor | patch

const pkg = JSON.parse(readFileSync(PKG_PATH, 'utf-8'));
const [major, minor, patch] = pkg.version.split('.').map(Number);

const bumpMap = {
  major: () => `${major + 1}.0.0`,
  minor: () => `${major}.${minor + 1}.0`,
  patch: () => `${major}.${minor}.${patch + 1}`,
};

if (!bumpMap[bumpType]) {
  console.error(`Usage: node bump-version.mjs `);
  process.exit(1);
}

const newVersion = bumpMap[bumpType]();
console.log(`Bumping: ${pkg.version} → ${newVersion} (${bumpType})`);

pkg.version = newVersion;
writeFileSync(PKG_PATH, JSON.stringify(pkg, null, 2) + '\n');

execSync(`git add ${PKG_PATH}`, { stdio: 'inherit' });
execSync(`git commit -m "chore: bump version to ${newVersion}"`, { stdio: 'inherit' });
execSync(`git tag v${newVersion} -m "Release v${newVersion}"`, { stdio: 'inherit' });

console.log(`Done. Run: git push origin main --tags && cd cdn && npm publish`);
