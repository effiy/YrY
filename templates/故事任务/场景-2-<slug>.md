### curl 命令

```bash
# GET /api/{{resource}} — {{用途}}
curl -s -w "\n%{http_code}" \
  -X GET \
  "${BASE_URL}/api/{{resource}}" \
  -H "Authorization: Bearer ${API_X_TOKEN}"

# 预期响应: HTTP 200
# { "items": [...], "total": N }

# POST /api/{{resource}} — {{用途}}
curl -s -w "\n%{http_code}" \
  -X POST \
  "${BASE_URL}/api/{{resource}}" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${API_X_TOKEN}" \
  -d '{"{{field}}":"{{value}}"}'

# 预期响应: HTTP 201
# { "id": "...", "{{field}}": "{{value}}" }
```
